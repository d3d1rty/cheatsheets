SELECT EMP_FNAME, EMP_LNAME
FROM EMPLOYEE
WHERE EMP_NUM NOT IN
(SELECT EMP_NUM FROM ASSIGNMENT)

SELECT J.JOB_CODE, JOB_DESCRIPTION, COUNT(*)
FROM JOB J, EMPLOYEE E
WHERE J.JOB_CODE = E.JOB_CODE
GROUP BY J.JOB_CODE, JOB_DESCRIPTION

SELECT P.PROJ_NUM,PROJ_NAME,SUM(ASSIGN_CHARGES)
FROM PROJECT P,ASSIGNMENT A
WHERE P.PROJ_NUM = A.PROJ_NUM
GROUP BY P.PROF_NUM,PROJ_NAME

FROM EMPLOYEE E,ASSIGNMENT A
WHERE E.EMP_NUM = A.EMP_NUM
GROUP BY E.EMP_NUM,EMP_FNAME,EMP_LNAME
