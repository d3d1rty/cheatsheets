--returns current date time
SELECT GETDATE()

--returns the month from the specified date
SELECT MONTH('5/22/2015')

--returns the month from the current date
SELECT MONTH(GETDATE())

--returns the year from the specified date
SELECT YEAR('5/22/2015')

--returns the year from the current date
SELECT YEAR (GETDATE())

--returns the day from the specified date
SELECT DAY('5/22/2015')

--returns the day from the current date
SELECT DAY(GETDATE())

--condition which specifies that the month of an attribute must be the fifth month
WHERE MONTH(ORDERDATE) = 5

--displays the month of the current date
SELECT DATEPART(MM,GETDATE())

--displays the year of the current date
SELECT DATEPART(YY,GETDATE())

--displays the name of the current day of the week
SELECT DATENAME(DW,GETDATE())
				
--returns the number of days between two dates
SELECT DATEDIFF(DD,'12/28/2014','1/3/2015')

--returns the date 8 days in the future from the current date
SELECT GETDATE() + 8

--returns the date 5 days in the future from the current date
SELECT DATEADD(DD,5,GETDATE())

--returns the date 30 days in the past from the current date
SELECT DATEADD(DD,-30,GETDATE())
